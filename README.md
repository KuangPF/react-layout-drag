## Install

```bash
npm install --save react-layout-drag
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-layout-drag'
import 'react-layout-drag/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
